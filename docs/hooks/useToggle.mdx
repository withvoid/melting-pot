# useToggle

```
const { on, onToggle } = useToggle();
```

## Examples

<Playground>
  {() => {
    const { on, onToggle } = useToggle();

    const styles = {
      container: {
        display: 'flex',
        flexDirection: 'row',
        justifyContent: 'center',
        backgroundColor: on ? '#81a1c1' : '#d8dee9',
        padding: 20,
      },
      button: {
        backgroundColor: on ? 'tomato' : 'gray',
        color: 'white',
        border: 0,
        padding: '10px 30px',
        outline: 0,
        cursor: 'pointer'
      },
    }

    return (
      <div style={styles.container}>
        <button style={styles.button} onClick={onToggle}>
          {on ? 'On' : 'Off'}
        </button>
      </div>
    );

}}

</Playground>
