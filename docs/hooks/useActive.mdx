# useActive

## Usage

```js
import { useActive } from '@withvoid/melting-pot';
```

## Syntax

```js
const { active, bind } = useActive();
```

## Examples

<Playground>
  {() => {
    const { active, bind } = useActive();
    const styles = {
      wrapper: {
        height: 300,
        display: 'flex',
        flexDirection: 'column',
        alignItems: 'center',
        justifyContent: 'center',
        cursor: 'pointer',
        backgroundColor: active ? 'red' : 'white',
        color: active ? 'white' : 'black',
      },
      emphasis: {
        display: 'inline-block',
        backgroundColor: active ? 'yellow' : 'red',
        color: active ? 'black' : 'white',
        padding: 5,
        width: 55,
        textAlign: 'center',
      },
    };
    const color = <span style={styles.emphasis}>{active ? 'red' : 'white'}</span>;
    return (
      <div {...bind} style={styles.wrapper}>
        <p>Click on the {color} background to change the active status</p>
        <p>
          Status: <span style={styles.emphasis}>{String(active)}</span>
        </p>
      </div>
    );
  }}
</Playground>

## API

useActive() returns a `boolean` `active`, which indicates if specified container
is in active state or not.
